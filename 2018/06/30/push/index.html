when all element in an Array([undefined,1,”1”]) is equal another([,1,”1”]) ,
but that two array was not equal ….
let arr = []
let len = arr.push(undefined, 1, “1”)
let arrNew = [, 1, “1”]
for (let i = 0; i < len; i++) {

    console.log(arr[i] === arrNew[i])//all true
}
console.log(arrNew === arr)//the result is false
